  class Solution {
  public:
      int divide(int dividend, int divisor) {
  		if (dividend == 0)
  			return 0;
  		if (divisor == 1) 
  			return dividend;			
  		if (divisor == -1) {
  			if (dividend == (-2147483647 - 1)) {
  				return 2147483647;
  			}				
  			return -dividend;
  		}			
  
  		int res = 0;
  		int flag = 1;
  		unsigned int udividend = dividend;
  		unsigned int udivisor = divisor;
  		if (dividend < 0) {
  			flag = -flag;
  			udividend = -dividend;
  		}
  		if (divisor < 0) {
  			flag = -flag;
  			udivisor = -divisor;
  		}
  
  		while (udividend >= udivisor) {
  			long long  i = udivisor;
  			int count = 1;
  			while (udividend >= (i<<1)) {
  				i = i << 1;
  				count =count<<1;
  			}
  			udividend -= i;
  			res += count;
  		}
  		if (flag < 0)
  			res = -res;
  		
  		return res;
      }
  };
